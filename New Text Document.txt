
#include <iostream>


using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
        int x,y;
        int count=0;
        cin>>x>>y;
        if((y%x)!=0){
            y++;
            count++;
        }else{
            count=0;
        }
        cout<<count<<endl;
    }
    

    return 0;
}
}
